1function [frameBER, estimate] = receiveOFDM80211a_sdru( tx, offsetCompensationValue, numFrames, useScopes, printReceivedData )
  2%#codegen
  3% receiveOFDM80211a: Recover OFDM signal from input data stream and 
  4% return synchronization estimates and BER values of that recovered data.
  5% Input signal is assumed to be based upon the IEEE 802.11a standard.
  6% Inputs:
  7%   -tx: struct containing transmitter's specifications
  8%   -offsetCompensationValue: additional carrier offset adjustment value
  9%   -numFrames: estimated number of frames in rChannel
 10%   -useScopes: enables scope visuals
 11%   -printPreceivedData: enables printing of recovered data to command window
 12
 13persistent ScopehArrayPilot  ScopehArrayPreamble ScopehSpect ScopehConstPre ScopehConstPost ScopehTimePeaks ScopehTimeFreqEst
 14coder.extrinsic('createOFDMScopes');%Ignore scopes when generating code
 15
 16%% Setup buffer parameters and system objects
 17% "buffer" is a window of samples from the input signal "recv".  This
 18% window is slid across the input signal in order to locate the start of
 19% frames. The variable "windowSlideDistance" determines the number of
 20% samples the buffer is slid between iterations. This sliding is not done
 21% on an individual sample basis for performance reasons. The "buffer"
 22% itself is set to be larger than a single frame to accommodate this group
 23% sliding, while still allowing for full individual frame detection.
 24
 25receiveBufferLength = ceil( tx.frameLength*4 ); %Size of Buffer of sliding window
 26
 27%Create memory structure to collect measurements for sync algorithms
 28numFreqToAverage = 15; %Number of frequency estimates to be averaged together for frequency corrections (Higher==More stability, Lower==More responsiveness)
 29[estimate, frameBER] = initializeOFDMSyncMemory_sdr( receiveBufferLength, numFreqToAverage, tx, printReceivedData ); 
 30
 31
 32% Create visualization objects
 33if useScopes
 34    [ScopehArrayPilot, ScopehArrayPreamble,...
 35     ScopehSpect, ScopehConstPre, ScopehConstPost,...
 36     ScopehTimePeaks, ScopehTimeFreqEst] = createOFDMScopes( tx, numFrames );
 37end
 38
 39% Setup USRP
 40DecimationFactor = 100e6/tx.samplingFreq;
 41hSDRu = comm.SDRuReceiver( '192.168.10.2', ...
 42    'CenterFrequency',      2.2e9 + offsetCompensationValue, ...
 43    'DecimationFactor',     DecimationFactor,...
 44    'FrameLength',          receiveBufferLength,...
 45    'OutputDataType',       'double',...
 46    'Gain',                 18);
 47
 48% Gain control
 49hAGC = comm.AGC('UpdatePeriod', receiveBufferLength);
 50
 51% Create demodulators objects
 52[ hPreambleDemod, hDataDemod ] = createOFDMDemods( tx );
 53
 54lastFound = -2; %Flag for found frame, used for dup check
 55numBuffersProcessed = 0; %Track received data, needed for separate indexing of processed and unprocessed data (processed==preamble found)
 56
 57totalBER = 0;
 58
 59%% Process received data
 60% Locate frames in buffer and compensate for channel affects
 61%for frames = 1 : numFrames
 62while estimate.numProcessed < numFrames    
 63    
 64    numBuffersProcessed = numBuffersProcessed + 1;% Increment processed data index
 65    
 66    %buffer = rChannel(windowIndex : windowIndex + estimate.inputBufferLength - 1);% Add incoming samples to buffer
 67    buffer = step(hSDRu);
 68    %if max(buffer)==0
 69    %    % All zeros from radio (Bug?)
 70    %    disp('All zeros (Bug?)');
 71    %    continue;
 72    %end
 73    buffer = step(hAGC, buffer);
 74    
 75    %% Find preamble in buffer
 76    [estimate.delay, estimate.numPeaks] = locateOFDMFrame_sdr( tx.FFTLength, tx.shortPreambleOFDM, buffer);
 77
 78    % Plot timing information  
 79    if useScopes
 80        step(ScopehTimePeaks, estimate.numPeaks);
 81    end
 82    
 83    % Check if frame exists in correct location and whether it's duplicate
 84    FrameFound = ((estimate.delay + tx.frameLength) < length(buffer) ) &&... %Check if full data frame exists in buffer
 85                 (estimate.delay > -1 ) &&... %Check if preamble located
 86                 ((numBuffersProcessed-lastFound) >= 2 ); %Check if duplicate frame
 87    
 88    %% Recover found frame
 89    if FrameFound
 90
 91        lastFound = numBuffersProcessed;%Flag frame as found so duplicate frames are not processed
 92        estimate.numProcessed = estimate.numProcessed + 1;%Increment processed found frames
 93        
 94        % Extract single frame from input buffer
 95        rFrame = buffer(estimate.delay + 1 : estimate.delay + tx.frameLength);        
 96               
 97        % Correct frequency offset
 98        [ rFreqShifted, estimate ] = coarseOFDMFreqEst_sdr( rFrame, tx, estimate);        
 99                
100        % Equalize
101        [ RPostEqualizer, RPreEqualizer, estimate] = equalizeOFDM( rFreqShifted, tx, estimate, hPreambleDemod, hDataDemod );
102        
103        % Demod subcarriers
104        [ frameBER, estimate ] = demodOFDMSubcarriers_sdr( RPostEqualizer, tx, estimate );
105        
106        % Visualize
107        if useScopes
108            stepOFDMScopes( ScopehArrayPilot, ScopehArrayPreamble,ScopehSpect,ScopehConstPre,...
109                            ScopehConstPost, ScopehTimeFreqEst, buffer, RPostEqualizer, RPreEqualizer, estimate )
110        end
111        %Just text
112        %if estimate.printReceivedData && ~useScopes
113         %   fprintf('Found frames: %i\n',int32(estimate.numProcessed));
114         %   fprintf('Averge frame BER: %f\n',frameBER);
115         %   %disp(['MSG: ', estimate.message.']);
116        %end
117    totalBER = totalBER + frameBER;
118        
119    end
120    
121   
122end
123
124%% Cleanup
125
126% Release objects
127if useScopes
128    release(ScopehArrayPilot);
129    release(ScopehArrayPreamble);
130    release(ScopehSpect);
131    release(ScopehConstPre);
132    release(ScopehConstPost);
133end
134
135release(hSDRu);
136
137
138% Print Results
139fprintf('Total BER: %f\n',totalBER/estimate.numProcessed);
140fprintf('Total received frames: %f\n',estimate.numProcessed);
141
142
143end
