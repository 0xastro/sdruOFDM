67% Calculate Equalizer Taps with pilot symbols
68function PilotGainsFull = pilotFDE(rxPilots, txPilots, upSampleFactor)
69
70% Calculate non-normalized channel gains
71PilotNormal = rxPilots./txPilots;
72
73% Scale channel gains

75PilotConj = conj(PilotNormal);
76PilotGains = PilotConj./ PilotEnergy;
77
78% Interpolate to data carrier size
