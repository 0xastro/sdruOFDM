21function [ rFreqShifted, estimate  ] = freqCorrectOFDM( tx, estimate, rFrame  )
22% freqCorrectOFDM: Apply frequency offset to input signal, this offset can
23% be generated through a moving average if enabled
24
25% Enable moving averages for estimates
26if estimate.numProcessed < estimate.numFreqToAverage % Buffer not full
27    
28    estimate.frequencyMA = mean(estimate.frequency(1:estimate.numProcessed));
29    
30    % Apply frequency correction
31    t = 0: 1/tx.samplingFreq : (length(rFrame)-1)/tx.samplingFreq;
32    rFreqShifted = exp(1i*estimate.frequencyMA*t.').*rFrame;
33    
34else % Full buffer
35    
36    estimate.frequencyMA = mean(estimate.frequency);
37    
38    % Apply frequency correction
39    t = 0: 1/tx.samplingFreq : (length(rFrame)-1)/tx.samplingFreq;
40    rFreqShifted = exp(1i*estimate.frequencyMA*t.').*rFrame;
41    
42end
43
44
45end
