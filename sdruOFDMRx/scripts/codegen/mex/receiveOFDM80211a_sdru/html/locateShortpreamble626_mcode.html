33function [preambleEstimatedLocation, numPeaks] = locateShortpreamble( M, K, threshold )
34%#codegen
35% Locate of the start of the actual preamble from timing metric
36
37%% Find peaks of correlation
38
39% Adjust threshold
40thresholdNorm = max(M)*threshold;
41MLocations = find(M>thresholdNorm);
42
43% Correct estimate to start of preamble not its center
44MLocations = MLocations - (K/2+1);
45
46% Frame Detection
47peaks = zeros(size(MLocations));
48
49% Determine correct peak 
50desiredPeakLocations = (16:16:128).';
51for i = 1:length(MLocations)
52    MLocationGuesses = MLocations(i)+desiredPeakLocations;
53    peaks(i) = length(intersect(MLocations(i:end),MLocationGuesses));
54end
55
56% Have at least 5 peaks for positive match
57requiredPeaks = 5; % (TUNABLE)
58peaks(peaks < requiredPeaks) = 0;
59
60% Pick earliest peak in time
61if ~isempty(peaks)
62    [numPeaks, frontPeakLocation] = max(peaks);
63    if ~isempty(frontPeakLocation) && ( numPeaks > 0 )
64        preambleEstimatedLocation = MLocations(frontPeakLocation);
65    else
66        preambleEstimatedLocation = -1; % no desirable location found
67    end
68else
69    preambleEstimatedLocation = -1;
70    numPeaks = 0;
71end
72% Normalize max peaks found
73numPeaks = numPeaks/numel(desiredPeakLocations);
74
75
76end
