1function [ rFreqShifted, estimate ] = coarseOFDMFreqEst_sdr( rFrame, tx, estimate)
 2%#codegen
 3% coarseOFDMFreqEst: Frequency Correction based on the Schmidl and Cox
 4% method, utilizing halves of the short preamble from the 802.11a standard
 5L=tx.FFTLength/2;
 6P = sum( conj(rFrame(1:L)  ) .* rFrame(1+L:L*2)  ); %Cross-correlate preamble
 7estimate.phi = angle(P); % Determine phase angle between preamble halves
 8
 9if estimate.numProcessed < estimate.numFreqToAverage% Buffer not full
10    estimate.frequency(estimate.numProcessed) = estimate.phi/(2*pi*L/tx.samplingFreq); % Convert angle to frequency
11else
12    estimate.frequency = [estimate.frequency(2:end);0];
13    estimate.frequency(end) = estimate.phi/(2*pi*L/tx.samplingFreq);
14end
15
16% Apply frequency estimate
17[ rFreqShifted, estimate  ] = freqCorrectOFDM( tx, estimate, rFrame  );
18
19end
